27static void fill_work_tree(work_tree_t *wt, list_t *list, const char *path)
28{
29 cell_t *tmp = NULL;
30 char *path_relative = NULL;
31
32 if (!wt || !list)
33 return;
34 tmp = *list;
35 for (size_t i = 0; i < wt->max_size && tmp; ++i) {
36 path_relative = get_path_absolute(path, tmp->data);
37 if (!path_relative)
38 return;
39 append_work_tree(wt, tmp->data, NULL, 0);
40 tmp = tmp->next;
41 free(path_relative);

45static void create_work_tree_and_iterate(work_tree_t *wt, work_file_t *wf,

61 fill_work_tree(wt2, list, path_absolute);
62 wt->tab[index].mode = get_chmod(path_absolute);
63 wt->tab[index].hash = save_content_and_work_tree(wt2, path);

65 free_list(list);

68char *save_content_and_work_tree(work_tree_t *wt, const char *path)
69{
70 work_file_t *wf = NULL;
71 char *path_absolute = NULL;
72
73 if (!wt || !path)
74 return NULL;
75 for (size_t i = 0; i < wt->size; ++i) {
76 wf = &wt->tab[i];
77 path_absolute = get_path_absolute(path, wf->name);
78 if (!path_absolute)
79 return NULL;
80 if (is_file(path_absolute))
81 save_content_file_get_mode_and_hash(wf, path_absolute);
82 else
83 create_work_tree_and_iterate(wt, wf, path, path_absolute, i);
84 free(path_absolute);
85 }
86 return create_blob_of_work_tree(wt);
87}
