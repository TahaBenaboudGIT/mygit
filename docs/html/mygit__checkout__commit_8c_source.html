34int mygit_checkout_commit(char *pattern)
35{
36 list_t *list_all_commits = NULL;
37 list_t *filtered_list = NULL;
38
39 if (!pattern)
40 return 0;
41 list_all_commits = get_list_all_commits();
42 filtered_list = filter_list_prefix(list_all_commits, pattern);
43 if (!list_all_commits || !*filtered_list) {
44 LOG_ERROR("Error: no commit found\n");
45 free_list(filtered_list);
46 free_list(list_all_commits);
47 return 0;
48 }
49 if ((*filtered_list)->next) {
50 print_error_several_commits_found(filtered_list);
51 free_list(filtered_list);
52 free_list(list_all_commits);
53 return 0;
54 }
55 checkout_commit_by_hash((*filtered_list)->data);
56 free_list(filtered_list);
57 free_list(list_all_commits);
58 return 1;
59}
