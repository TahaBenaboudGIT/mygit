59// Due to a non support in MacOS of %ms, in sscanf
60// we use %s and we allocate the memory
61int get_content_str_work_file(const char *str, char **name, char **hash,
62 mode_t *mode)
63{
64 if (!name || !hash || !mode || !str)
65 return 0;
66 *name = malloc(BUFFER_SIZE * sizeof(char));
67 if (!*name)
68 return 0;
69 *hash = malloc(BUFFER_SIZE * sizeof(char));
70 if (!*hash) {
71 free(*name);
72 return 0;
73 }
74 if (sscanf(str, "%s\t%s\t%ho", *name, *hash, mode) < 2)
75 return 0;
76 set_to_null_if_dash(name);
77 set_to_null_if_dash(hash);
78 return 1;
79}
